Учебник JavaScript

Типы данных

alert, prompt, confirm: Функции для взаимодействия с пользователем через модальные окна.

  • alert(message): Отображает модальное окно с сообщением message.
  • prompt(message, default): Отображает модальное окно с сообщением message и текстовым полем для ввода данных, предварительно заполненным значением default.
  • confirm(message): Отображает модальное окно с сообщением message и кнопками "ОК" и "Отмена", возвращая true при нажатии "ОК" и false при нажатии "Отмена".

 
// Пример:
alert('Привет, мир!');
let name = prompt('Как тебя зовут?', 'Имя');
let isAdmin = confirm('Ты администратор?');

Переменные

Переменные в JavaScript объявляются с использованием ключевых слов var, let или const.

  • var: Глобальная или функциональная область видимости.
  • let: Блочная область видимости, позволяет изменять значение переменной.
  • const: Блочная область видимости, значение которой не может быть изменено после инициализации.


// Пример:
var age = 30;
let userName = 'John';
const pi = 3.14;

console.log

console.log используется для вывода сообщений в консоль разработчика браузера. Это помогает отслеживать выполнение программы и отладку кода. 


// Пример:
console.log('Это сообщение будет выведено в консоль');

Условные операторы if/else

Условные операторы if и else используются для выполнения кода, если определенное условие истинно (if) или ложно (else). 


// Пример:
let number = 10;
if (number > 5) {
    console.log('Число больше 5');
} else {
    console.log('Число меньше или равно 5');
}

Логические операторы || и ??

Операторы || и ?? используются для работы с логическими значениями и возвращают первый истинный операнд или последний операнд, если все ложные. 


// Пример:
let name = null;
let defaultName = 'Гость';
let userName = name || defaultName;
console.log(userName); // 'Гость'

let user = null;
let guest = 'Гость';
let currentUser = user ?? guest;
console.log(currentUser); // 'Гость'

Циклы for и while

for и while используются для многократного выполнения блока кода. 


// Пример:
for (let i = 0; i < 5; i++) {
    console.log(i);
}

let j = 0;
while (j < 5) {
    console.log(j);
    j++;
}

Функции

Функции в JavaScript являются блоками кода, предназначенными для выполнения определенных задач. Они могут принимать аргументы и возвращать значения. 


// Пример:
function greet(name) {
    return `Привет, ${name}!`;
}

console.log(greet('Вася'));

Оператор switch

switch используется для выбора выполнения кода из множества вариантов на основе значения выражения. 


// Пример:
let color = 'green';

switch (color) {
    case 'red':
        console.log('Стоп');
        break;
    case 'yellow':
        console.log('Внимание');
        break;
    case 'green':
        console.log('Иди');
        break;
    default:
        console.log('Неизвестный цвет');
}

Комментарии

Комментарии в JavaScript используются для объяснения кода и игнорируются интерпретатором при выполнении программы. 


// Это однострочный комментарий

/*
  Это
  многострочный
  комментарий
*/

Отладчик (debugger)

debugger используется для остановки выполнения JavaScript кода и перехода в режим отладки в консоли разработчика браузера. 


// Пример:
function testDebug() {
    debugger; // Выполнение остановится здесь
    console.log('Отладка');
}

testDebug();

Объекты

Объекты в JavaScript это сложные структуры данных, которые содержат ключ-значение и методы для работы с данными. 


// Пример:
let person = {
    name: 'John',
    age: 30,
    greet: function() {
        console.log('Привет, ' + this.name);
    }
};

person.greet();

Контекст (this)

Ключевое слово this в JavaScript ссылается на текущий объект, в контексте которого выполняется текущий код. 


// Пример:
let user = {
    name: 'Alice',
    sayHi() {
        console.log(this.name);
    }
};

user.sayHi();

Методы Object

Object.keys(), Object.values(), и Object.entries() используются для работы с объектами и их свойствами. 


// Пример:
let person = {
    name: 'Bob',
    age: 25
};

console.log(Object.keys(person)); // ['name', 'age']
console.log(Object.values(person)); // ['Bob', 25]
console.log(Object.entries(person)); // [['name', 'Bob'], ['age', 25]]

Дата и время

new Date() используется для работы с датами и временем в JavaScript. 


// Пример:
let now = new Date();
console.log(now);

Рекурсия

Рекурсия это процесс, когда функция вызывает саму себя для решения задачи. 


// Пример:
function factorial(n) {
    if (n === 0) {
        return 1;
    }
    return n * factorial(n - 1);
}

console.log(factorial(5)); // 120

Замыкания

Замыкания это функции, которые имеют доступ к переменным из внешней функции, даже после завершения выполнения этой внешней функции. 


// Пример:
function makeCounter() {
    let count = 0;
    return function() {
        return count++;
    };
}

let counter = makeCounter();
console.log(counter()); // 0
console.log(counter()); // 1

setTimeout и setInterval

setTimeout и setInterval используются для выполнения функций с задержкой и через определенные интервалы времени соответственно. 


// Пример:
setTimeout(() => {
    console.log('Прошло 2 секунды');
}, 2000);

let intervalId = setInterval(() => {
    console.log('Прошла 1 секунда');
}, 1000);

setTimeout(() => {
    clearInterval(intervalId);
}, 5000);

Остаточные параметры и оператор расширения

Остаточные параметры ... позволяют функции принимать неограниченное количество аргументов, а оператор расширения ... используется для развертывания массивов или объектов. 


// Пример:
function sum(...args) {
    return args.reduce((acc, value) => acc + value, 0);
}

console.log(sum(1, 2, 3, 4)); // 10

let arr = [1, 2, 3];
let newArr = [...arr, 4, 5];
console.log(newArr); // [1, 2, 3, 4, 5]

Прототип (_proto_)

Прототип в JavaScript это механизм наследования объектов, который позволяет объектам наследовать свойства и методы других объектов. 


// Пример:
let animal = {
    eats: true
};

let rabbit = {
    jumps: true,
    __proto__: animal
};

console.log(rabbit.eats); // true

Массивы

Массивы в JavaScript представляют собой упорядоченную коллекцию элементов, индексируемых числовыми индексами. 


// Пример:
let fruits = ['Apple', 'Banana', 'Cherry'];
console.log(fruits[1]); // 'Banana'

BOM и DOM

BOM (Browser Object Model) и DOM (Document Object Model) представляют собой объектные модели браузера и документа, используемые для взаимодействия с окружением браузера и веб-страницами. 


// Пример:
document.body.style.backgroundColor = 'lightblue';
console.log(window.innerWidth);

Fetch API

Fetch API используется для отправки и получения данных через сеть, заменяя устаревший XMLHttpRequest. 


// Пример:
fetch('https://jsonplaceholder.typicode.com/todos/1')
    .then(response => response.json())
    .then(json => console.log(json));

Cookie и Local Storage

Cookie и Local Storage являются механизмами хранения данных на стороне клиента в браузере. 


// Пример:
document.cookie = 'username=John';
localStorage.setItem('user', 'John');

console.log(document.cookie);
console.log(localStorage.getItem('user'));

Регулярные выражения

Регулярные выражения (RegExp) используются для поиска и замены текста на основе шаблона. 


// Пример:
let re = /\d+/g;
let str = 'Есть 100 яблок и 20 бананов';
console.log(str.match(re)); // ['100', '20']

Промисы

Промисы (Promise) в JavaScript используются для управления асинхронными операциями и их результатами. 


// Пример:
let promise = new Promise((resolve, reject) => {
    setTimeout(() => resolve('Успех!'), 2000);
});

promise.then(result => console.log(result)); // 'Успех!'

Колбэки

Колбэки (Callback) в JavaScript это функции, передаваемые в качестве аргументов другим функциям и вызываемые после завершения определенной задачи. 


// Пример:
function loadScript(src, callback) {
    let script = document.createElement('script');
    script.src = src;
    script.onload = () => callback(script);
    document.head.append(script);
}

loadScript('path/to/script.js', (script) => {
    console.log('Скрипт загружен: ' + script.src);
});

Асинхронность и await

Асинхронные функции и ключевое слово await используются для работы с асинхронными операциями в JavaScript в более читаемом и удобном виде. 


// Пример:
async function fetchData() {
    let response = await fetch('https://jsonplaceholder.typicode.com/todos/1');
    let data = await response.json();
    console.log(data);
}

fetchData();

Всплытие и погружение

Всплытие (Bubbling) и погружение (Capturing) относятся к порядку обработки событий в DOM веб-страницы. 


// Пример:
document.getElementById('myDiv').addEventListener('click', () => {
    console.log('Clicked!');
}, true); // Погружение

document.getElementById('myDiv').addEventListener('click', () => {
    console.log('Clicked!');
}, false); // Всплытие

JSON

JSON (JavaScript Object Notation) используется для обмена данными между сервером и клиентом в удобном для чтения формате. 


// Пример:
let jsonString = '{"name": "John", "age": 30}';
let user = JSON.parse(jsonString);
console.log(user.name); // 'John'

let json = JSON.stringify(user);
console.log(json); // '{"name":"John","age":30}'

Методы массивов

Методы массивов в JavaScript предоставляют множество функций для работы с массивами, включая добавление, удаление, фильтрацию и преобразование элементов. 


// Пример:
let numbers = [1, 2, 3, 4, 5];

let doubled = numbers.map(num => num * 2);
console.log(doubled); // [2, 4, 6, 8, 10]

let evens = numbers.filter(num => num % 2 === 0);
console.log(evens); // [2, 4]

numbers.push(6);
console.log(numbers); // [1, 2, 3, 4, 5, 6]

Остальные операторы

В JavaScript также есть различные операторы, такие как арифметические, логические, побитовые и т.д., которые позволяют выполнять операции над значениями и переменными. 


// Пример:
let x = 10;
let y = 5;

console.log(x + y); // 15
console.log(x - y); // 5
console.log(x * y); // 50
console.log(x / y); // 2

console.log(x > y); // true
console.log(x < y); // false
console.log(x === y); // false

let z = x & y; // Побитовый И
console.log(z); // 0
